from django.contrib.auth.forms import UserCreationForm from django.contrib.auth.models import User from django.shortcuts import render, redirect def signup(request): if request.method == "POST": form = UserCreationForm(request.POST) if form.is_valid(): form.save() return redirect("login") # Redirect to login page after signup else: form = UserCreationForm() return render(request, "signup.html", {"form": form})
